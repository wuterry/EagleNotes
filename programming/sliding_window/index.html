    def lengthOfLongestSubstring(self, s: str) -> int:
    max_len = 0
    i = 0
    hashmap = dict()
    for j in enumerate(len(s)):
        # update (hashmap, max_len)
        hashmap[s[j]] = hashmap.get(s[j], 0) + 1
        if len(hashmap) <= 2:
            curr_len = j - i + 1
            max_len = max(max_len, curr_len)

        # æ›´æ–°å·¦æŒ‡é’ˆ

            hashmap[s[i]] -= 1
            if hashmap[s[i]] == 0:
                del hashmap[s[i]]
            i += 1
    return max_len

    def maxScore(self, cardPoints: List[int], k: int) -> int:

    win_size = n - k
    min_val = float('inf')

    # ç‰¹ä¾‹

    i = 0
    win_sum = 0  # sum of window
    for j in range(n):
