    def lengthOfLongestSubstring(self, s: str) -> int:
    """快慢指针
    1. 使用集合记录窗的大小;
    2. 当遍历的字符在窗中未出现过，则快指针 +1;
    3. 当遍历的字符在窗中已经存在，则慢指针 +1.
    """
    max_len = 0
    i = 0
    buffer = set()
    for j, char in enumerate(s):
        # 移动左指针
        while char in buffer:
            buffer.remove(s[i])
            i += 1
        # 移动右指针
        buffer.add(char)
        # 更新最大值
        max_len = max(max_len, len(buffer))
    return max_len

    def lengthOfLongestSubstring(self, s: str) -> int:
    max_len = 0
    i = 0
    hashmap = dict()
    for j in enumerate(len(s)):
        # update (hashmap, max_len)
        hashmap[s[j]] = hashmap.get(s[j], 0) + 1
        if len(hashmap) <= 2:
            curr_len = j - i + 1
            max_len = max(max_len, curr_len)

        # 更新左指针
        while len(hashmap) > 2:
            hashmap[s[i]] -= 1
            if hashmap[s[i]] == 0:
                del hashmap[s[i]]
            i += 1
    return max_len

    def isin(cnt_t, cnt_w):
    """验证 cnt_t 是否每个元素的个数都不大于 cnt_w """
    for k, v in cnt_t.items():
        if v > cnt_w[k]:
            return False
    return True

    def minSubArrayLen(self, target: int, nums: List[int]) -> int:
    min_len = len(nums) + 1
    i = 0
    sum = 0
    for j in range(len(nums)):
        sum += nums[j]

        while sum >= target:
            min_len = min(min_len, j - i + 1)
            sum -= nums[i]
            i += 1
    return 0 if min_len > len(nums) else min_len

    def longestOnes(self, nums: List[int], k: int) -> int:
    counter_win = {1: 0, 0: 0}
    max_len = 0
    i = 0
    for j, num in enumerate(nums):
        counter_win[num] = counter_win.get(num, 0) + 1
        if counter_win[0] <= k:
            max_len = max(max_len, j - i + 1)

        while counter_win[0] > k:
            counter_win[nums[i]] -= 1
            i += 1
    return max_len

    def findAnagrams(self, s: str, p: str) -> List[int]:
    counter_p = dict()
    for c in p:
        counter_p[c] = counter_p.get(c, 0) + 1
    counter_win = dict()

    ans = []
    i = 0
    for j in range(len(s)):
        counter_win[s[j]] = counter_win.get(s[j], 0) + 1
        if counter_p == counter_win:
            ans.append(i)

        if j - i + 1 >= len(p):  # 注意这里取等号
            counter_win[s[i]] -= 1
            if counter_win[s[i]] == 0:
                del counter_win[s[i]]
            i += 1
    return ans

    def maxScore(self, cardPoints: List[int], k: int) -> int:
    n = len(cardPoints)
    win_size = n - k
    min_val = float('inf')

    # 特例
    if n == k:
        return sum(cardPoints)

    i = 0
    win_sum = 0  # sum of window
    for j in range(n):
        win_sum += cardPoints[j]

        if j - i + 1 >= win_size:
            min_val = min(min_val, win_sum)
            win_sum -= cardPoints[i]
            i += 1

    return sum(cardPoints) - min_val

    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
    """单调队列法
    设定一个双向队列 deque, 保存元素索引，索引对应的元素值始终为递减数列
    Hints: 队列中没必要维护靠前的元素值小于当前元素值的索引。
    入队：
        1. 先删后插：当队尾元素 deque[-1] <= nums[i], 则先 pop, 再追加
        2. 直接插入：当队尾元素 deque[-1] > nums[i], 则直接追加
    出队：
        当 deque[0] 滑出窗外
    """
    results = []
    q = collections.deque()

    for i in range(len(nums)):
        # 队尾入队
        while q and nums[i] >= nums[q[-1]]:
            q.pop()
        q.append(i)

        # 队首出队: 队首元素超出窗范围
        if q[0] <= i - k:
            q.popleft()

        # 保存元素
        if i >= k - 1:
            results.append(nums[q[0]])
    return results
