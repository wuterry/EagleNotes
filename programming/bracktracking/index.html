def backtracking(路径, 选择列表):
    if 满足结束条件:

        撤销选择

    题目: 给定一个 m x n 二维字符网格 board 和一个字符串单词 word 。如果 word 存在于网格中，返回 true ；否则，返回 false 。 单词必须按照字母顺序，通过相邻的单元格内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。同一个单元格内的字母不允许被重复使用。

    def exist(self, board: List[List[str]], word: str) -> bool:
    self.flag = False  # 定义一个全局状态

    def backtrack(i, j, start, visited):
        # 越界或已遍历
        if i < 0 or j < 0 or i >= m or j >= n or visited[i][j]:
            return
        # 字符不匹配
        if board[i][j] != word[start]:
            return
        if start == len(word) - 1:
            self.flag = True
            return

        visited[i][j] = True  # 选择
        backtrack(i - 1, j, start + 1, visited)
        backtrack(i + 1, j, start + 1, visited)
        backtrack(i, j - 1, start + 1, visited)
        backtrack(i, j + 1, start + 1, visited)
        visited[i][j] = False  # 撤销选择

    m, n = len(board), len(board[0])
    # 定义是否已遍历的数组
    visited = [[False for j in range(n)] for i in range(m)]
    for i in range(m):
        for j in range(n):
            start = 0
            self.flag = False
            backtrack(i, j, start, visited)
            if self.flag:
                return True
    return False

    51.N皇后#

    题目: 按照国际象棋的规则，皇后可以攻击与之处在同一行或同一列或同一斜线上的棋子。 n 皇后问题 研究的是如何将 n 个皇后放置在 n×n 的棋盘上，并且使皇后彼此之间不能相互攻击。 给你一个整数 n ，返回所有不同的 n 皇后问题 的解决方案。 每一种解法包含一个不同的 n 皇后问题 的棋子放置方案，该方案中 'Q' 和 '.' 分别代表了皇后和空位。

    def solveNQueens(self, n: int) -> List[List[str]]:
    def generate_board():
        board = []
        for r in range(n):
            row[queens[r]] = "Q"
            board.append("".join(row))
            row[queens[r]] = "."  # 恢复为 "."
        return board

    def backtrack(r):
        """
        r (int): index of row
        """
        if r == n:
            board = generate_board()
            results.append(board)
            return

        for c in range(n):
            if (c in columns) or (r - c in diag_45) or (r + c in diag_135):
                continue
            queens[r] = c
            columns.add(c)
            diag_45.add(r - c)
            diag_135.add(r + c)
            backtrack(r + 1)
            columns.remove(c)
            diag_45.remove(r - c)
            diag_135.remove(r + c)

    results = []
    queens = [-1] * n  # store colomn index of queens in each row
    row = ["."] * n
    columns = set()
    diag_45 = set()
    diag_135 = set()

    backtrack(r=0)
    return results

    def letterCombinations(self, digits: str) -> List[str]:
    NUMBER_MAP = {
        "2": "abc",
        "3": "def",
        "4": "ghi",
        "5": "jkl",
        "6": "mno",
        "7": "pqrs",
        "8": "tuv",
        "9": "wxyz"
    }

    def backtrack(digits, index, track, results):
        if len(digits) == len(track):
            results.append("".join(track))
            return

        digit = digits[index]
        for letter in NUMBER_MAP[digit]:
            track.append(letter)
            backtrack(digits, index + 1, track, results)
            track.pop()

    if not digits:
        return []
    results = []
    track = []
    index = 0
    backtrack(digits, index, track, results)
    return results

    def generateParenthesis(self, n: int) -> List[str]:
    def backtrack(left, right, track, results):
        """
        Args:
            left (int): 左括号的个数
            right (int): 右括号的个数
            track (list): 路径
            results (list[list]): 保存路径的列表
        注意: 剪枝和添加结果的顺序不能交换
        """
        # 剪枝
        if right > left or left > n or right > n:
            return

        # 满足条件则添加到结果列表
        if len(track) == 2 * n:
            results.append("".join(track))
            return

        # 增加左括号
        track.append("(")
        backtrack(left + 1, right, track, results)
        track.pop()

        # 增加右括号
        track.append(")")
        backtrack(left, right + 1, track, results)
        track.pop()

    results = []
    track = []
    if n == 0:
        return results
    backtrack(0, 0, track, results)
    return results

    def generateParenthesis(self, n: int) -> List[str]:
    def dfs(curr_str, left, right, results):
        if left > n or right > n or right > left:
            return
        if len(curr_str) == n * 2:
            results.append(curr_str)
            return
        dfs(curr_str + "(", left + 1, right, results)
        dfs(curr_str + ")", left, right + 1, results)

    string = ""
    results = []
    dfs(string, 0, 0, results)
    return results

    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
    def backtrack(track_sum, track, results, start):
        """
        track_sum (int): sum of track
        start (int): the start index for start
        """
        if track_sum > target:
            return
        if track_sum == target:
            results.append(track.copy())  # 注意使用 copy()
            return

        for i in range(start, len(candidates)):
            # 选择并处理
            track_sum += candidates[i]
            track.append(candidates[i])
            # 回溯
            backtrack(track_sum, track, results, i)
            # 撤销选择
            track_sum -= candidates[i]
            track.pop()

    results = []
    track = []
    track_sum = 0
    start = 0
    backtrack(track_sum, track, results, start)
    return results

    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
    def backtrack(track, track_sum, results, start):
        if track_sum > target:
            return
        if track_sum == target:
            results.append(track.copy())
            return

        for i in range(start, len(candidates)):
            # 去重
            if i > start and candidates[i - 1] == candidates[i]:
                continue

            track.append(candidates[i])
            track_sum += candidates[i]
            backtrack(track, track_sum, results, i + 1)
            track.pop()
            track_sum -= candidates[i]

    results = []
    track = []
    track_sum = 0
    start = 0

    candidates = sorted(candidates)  # 先排序
    backtrack(track, track_sum, results, start)
    return results

    def permute(self, nums: List[int]) -> List[List[int]]:
    results = []
    track = []
    self.backtrack(nums, track, results)
    return results

def backtrack(self, nums, track, results):
    if len(track) == len(nums):
        results.append(track.copy())

    for num in nums:
        if num in track:  # 当前层中, 已添加的数不再考虑
            continue
        track.append(num)                     # 选择
        self.backtrack(nums, track, results)  # 递归

    def subsets(self, nums: List[int]) -> List[List[int]]:
    def backtrack(track, results, start):
        if len(track) > len(nums):
            return
        results.append(track[:])

        for i in range(start, len(nums)):
            num = nums[i]
            if num in track:
                continue
            track.append(num)
            backtrack(track, results, i)
            track.pop()

    results = []
    track = []
    backtrack(track, results, 0)
    return results
