    def partition(arr, low, high):
    # rightmost element as pivot
    pivot = arr[high]
    l, r = low, high

    while l < r:
        while arr[l] <= pivot and l < r:
            l += 1
        while arr[r] >= pivot and l < r:
            r -= 1
        arr[l], arr[r] = arr[r], arr[l]
    arr[l], arr[high] = arr[high], arr[l]
    return l

def quick_sort(arr, low, high):
    if low < high:
        # find the index of pivot element 
        pi = partition(arr, low, high)
        quick_sort(arr, low, pi - 1)
        quick_sort(arr, pi + 1, high)

quick_sort(arr, 0, len(arr) - 1)

    def merge(arr1, arr2):
    # merge sorted array arr1 and arr2
    i, j = 0, 0
    result = []
    while i < len(arr1) and j < len(arr2):
        if arr1[i] <= arr2[j]:
            result.append(arr1[i])
            i += 1
        else:
            result.append(arr2[j])
            j += 1
    result += arr1[i:]
    result += arr2[j:]
    return result


def merge_sort(nums):
    if len(nums) <= 1:
        return nums
    mid = len(nums) // 2

    arr1 = merge_sort(nums[:mid])
    arr2 = merge_sort(nums[mid:])
    merged = merge(arr1, arr2)
    return merged

    def heapify(nums, n, i):
    if i >= n:
        return

    max_ = i  # initialize max as root
    l = 2 * i + 1
    r = 2 * i + 2

    # if child exists and is greater than root, exchange
    if l < n and nums[max_] < nums[l]:
        max_ = l
    if r < n and nums[max_] < nums[r]:
        max_ = r

    # change root if need
    if max_ != i:
        nums[i], nums[max_] = nums[max_], nums[i]  # swap
        # heapify the root
        heapify(nums, n, max_)


def build_heap(nums, n):
    # build from the last second layer
    parent = (n - 1) // 2
    for i in range(parent, -1, -1):
        heapify(nums, n, i)

def heap_sort(nums):
    n = len(nums)

    # build a max heap
    build_heap(nums, n)

    # extract elements one by one
    for i in range(n - 1, -1, -1):
        nums[i], nums[0] = nums[0], nums[i]  # swap
        heapify(nums, i, 0)

    def findKthLargest(self, nums: List[int], k: int) -> int:
    import random

    def partition(nums, left, right):
        """æ„é€ é™åºæ•°ç»„ï¼Œå·¦å¤§å³å°"""
        # éšæœºé€‰å–ä¸€ä¸ªå…ƒç´ å¹¶äºç¬¬ä¸€ä¸ªå…ƒç´ äº¤æ¢, é˜²æ­¢å¿«æ’é€€åŒ–åˆ° O(n^2)
        rand_idx = random.randint(left, right)
        nums[left], nums[rand_idx] = nums[rand_idx], nums[left]

        pivot = nums[left]
        i, j = left, right
        while i < j:
            while nums[j] <= pivot and i < j:
                j -= 1
            while nums[i] >= pivot and i < j:
                i += 1
            nums[i], nums[j] = nums[j], nums[i]
        nums[i], nums[left] = pivot, nums[i]
        return i

    left, right = 0, len(nums) - 1
    while left < right:
        index = partition(nums, left, right)
        if index == k - 1:
            return nums[index]
        elif index < k - 1:
            left = index + 1
        else:
            right = index - 1
    return nums[left]

    def findKthLargest(self, nums: List[int], k: int) -> int:
    def heapify(nums, n, i):
        largest = i
        lchild = i * 2 + 1
        rchild = i * 2 + 2
        if lchild < n and nums[lchild] > nums[largest]:
            largest = lchild
        if rchild < n and nums[rchild] > nums[largest]:
            largest = rchild
        if largest != i:
            nums[largest], nums[i] = nums[i], nums[largest]
            heapify(nums, n, largest)

    def build_heap(nums, n):
        for i in range((n - 1) // 2, -1, -1):
            heapify(nums, n, i)

    n = len(nums)
    build_heap(nums, n)
    for i in range(n - 1, n - k - 1, -1):
        nums[i], nums[0] = nums[0], nums[i]
        heapify(nums, i, 0)
    return nums[n - k]
